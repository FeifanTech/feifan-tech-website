案例一

商家销量预测

时序预测 Prophet LightGBM 集成学习
业务背景

飞凡平台上有超过5000个活跃商家，备货决策主要依赖商家经验。库存不足导致缺货损失每月约15%，库存积压导致的资金占用和损耗每月约8%。通过销量预测系统，帮助商家提前7~30天进行科学备货。

问题定义
预测目标SKU日销售量（件数）
预测粒度SKU × 日期
预测窗口未来7天/30天
评估指标MAPE < 20%
运行方式每日凌晨批量
数据描述
数据源字段说明质量备注
销售记录表sku_id, date, sales_qty, gmv2年历史，每日聚合约2%缺失（门店关闭日）
商品属性表category, brand, price_tier静态属性质量较好
营销活动表promo_type, discount_rate, start_date, end_date促销记录历史活动覆盖率80%
节假日表date, holiday_name, is_holiday中国节假日+电商大促完整
算法选择与理由
采用 Prophet + LightGBM 集成方案
  • 📌 Prophet：分解趋势、周效应、年效应、节假日效应，输出组件级解释，商家可理解
  • 📌 LightGBM：捕捉促销、价格等外生特征与销量的非线性关系
  • 📌 集成方式：Prophet预测作为特征输入LightGBM，加权平均最终预测
  • 📌 新品处理：历史<30天的SKU用同类目均值降级
关键代码实现
Python Prophet + LightGBM 销量预测集成 
from prophet import Prophet
import lightgbm as lgb
import pandas as pd
import numpy as np

class SalesForecastEnsemble:
    def __init__(self, horizon: int = 7):
        self.horizon = horizon
        self.prophet_models = {}
        self.lgb_model = None

    def _fit_prophet(self, df_sku: pd.DataFrame) -> Prophet:
        """为单个SKU训练Prophet模型"""
        m = Prophet(
            yearly_seasonality=True,
            weekly_seasonality=True,
            daily_seasonality=False,
            changepoint_prior_scale=0.05
        )
        # 添加中国节假日
        m.add_country_holidays(country_name='CN')
        train = df_sku[['date', 'sales_qty']].rename(
            columns={'date': 'ds', 'sales_qty': 'y'})
        m.fit(train)
        return m

    def get_prophet_features(self, sku_id: str, future_dates: pd.DataFrame) -> np.ndarray:
        """提取Prophet预测作为LightGBM特征"""
        model = self.prophet_models.get(sku_id)
        if model is None:
            return np.zeros(len(future_dates))
        forecast = model.predict(future_dates)
        return forecast['yhat'].clip(lower=0).values

    def fit(self, df: pd.DataFrame):
        """训练全部SKU的Prophet和集成LightGBM"""
        # 1. 为每个SKU训练Prophet（批量，耗时约10分钟）
        for sku_id, group in df.groupby('sku_id'):
            if len(group) >= 60:  # 至少60天历史才训练
                self.prophet_models[sku_id] = self._fit_prophet(group)

        # 2. 为LightGBM构建训练集（含Prophet预测作为特征）
        df['prophet_pred'] = df.apply(
            lambda r: self.prophet_models.get(r['sku_id'], None), axis=1
        )  # 简化示意，实际需逐SKU调用predict

        feature_cols = ['prophet_pred', 'dayofweek', 'month',
                        'is_holiday', 'promo_type', 'price',
                        'sales_lag_7', 'sales_roll_mean_28']

        self.lgb_model = lgb.LGBMRegressor(
            n_estimators=500, learning_rate=0.05,
            num_leaves=63, objective='mape'
        )
        return self

关键代码实现
Python 矩阵分解训练用户/商品向量 
import implicit
import faiss
import scipy.sparse as sp
import numpy as np

# ── 构建用户-商品交互矩阵 ──
# 使用隐式反馈（浏览/点击/收藏/购买权重不同）
def build_interaction_matrix(events_df):
    weights = {'view': 1, 'click': 3, 'cart': 5, 'purchase': 10}
    events_df['weight'] = events_df['event_type'].map(weights)
    pivot = events_df.groupby(['user_idx', 'item_idx'])['weight'].sum().reset_index()
    return sp.csr_matrix(
        (pivot['weight'], (pivot['user_idx'], pivot['item_idx']))
    )

# ── 训练ALS矩阵分解 ──
model = implicit.als.AlternatingLeastSquares(
    factors=64, iterations=20, regularization=0.01
)
model.fit(interaction_matrix)

user_factors = model.user_factors  # shape: (n_users, 64)
item_factors = model.item_factors  # shape: (n_items, 64)

# ── 构建Faiss索引（近似最近邻检索）──
index = faiss.IndexFlatIP(item_factors.shape[1])  # 内积相似度
faiss.normalize_L2(item_factors)  # 归一化 → IP等价于余弦相似度
index.add(item_factors)

# ── 在线召回（<10ms）──
def recall_for_user(user_id: int, k: int = 500) -> np.ndarray:
    user_vec = user_factors[user_id].reshape(1, -1)
    faiss.normalize_L2(user_vec)
    _, item_indices = index.search(user_vec, k)
    return item_indices[0]  # 返回Top-k商品ID

关键代码实现
Python 规则引擎 + 决策树控制器 
from sklearn.tree import DecisionTreeClassifier
from dataclasses import dataclass
from typing import Optional
import joblib

@dataclass
class SensorReading:
    temperature: float   # 水温 °C
    dissolved_o2: float  # 溶氧 mg/L
    ph: float            # pH值
    ammonia: float       # 氨氮 mg/L

class AquacultureController:
    # 安全阈值（硬约束，不可绕过）
    SAFETY_RULES = {
        'dissolved_o2_critical': 4.0,  # 低于此值立即开最大增氧
        'temperature_max': 32.0,        # 超过此温度停止投料
        'ph_low': 6.5,                  # pH过低触发告警
        'ph_high': 9.0,
    }

    def __init__(self, dt_model_path: str):
        self.dt_model = joblib.load(dt_model_path)

    def _safety_check(self, reading: SensorReading) -> Optional[dict]:
        """安全规则优先级最高，有触发立即执行"""
        rules = self.SAFETY_RULES
        if reading.dissolved_o2 < rules['dissolved_o2_critical']:
            return {'aerator': 'MAX', 'feeder': 'STOP', 'alert': 'CRITICAL_DO'}
        if reading.temperature > rules['temperature_max']:
            return {'feeder': 'STOP', 'alert': 'HIGH_TEMP'}
        return None  # 无安全问题，进入决策树优化

    def decide(self, reading: SensorReading) -> dict:
        """控制决策主入口"""
        safety_action = self._safety_check(reading)
        if safety_action:
            return safety_action  # 安全规则优先

        # 决策树优化控制（软策略）
        features = [[reading.temperature, reading.dissolved_o2,
                     reading.ph, reading.ammonia]]
        action_code = self.dt_model.predict(features)[0]
        return self._decode_action(action_code)

    def _decode_action(self, code: int) -> dict:
        actions = {
            0: {'aerator': 'LOW',  'feeder': 'NORMAL'},
            1: {'aerator': 'MED',  'feeder': 'NORMAL'},
            2: {'aerator': 'HIGH', 'feeder': 'REDUCE'},
        }
        return actions.get(code, actions[1])

# 微调训练命令（在标注数据上）：
# model.train(data='fish_dataset.yaml', epochs=50, imgsz=640, batch=16)

class FishBehaviorMonitor:
    def __init__(self, model_path: str, window: int = 30):
        self.model = YOLO(model_path)
        self.history = deque(maxlen=window)  # 30帧历史
        self.alert_threshold = 2.5  # 密度标准差阈值

    def process_frame(self, frame: np.ndarray) -> dict:
        results = self.model(frame, conf=0.45, verbose=False)[0]
        boxes = results.boxes

        metrics = {
            'fish_count': len(boxes),
            'avg_confidence': float(boxes.conf.mean()) if len(boxes) > 0 else 0,
        }

        # 计算鱼群质心和密度分布
        if len(boxes) > 5:
            centers = boxes.xywh[:, :2].cpu().numpy()
            metrics['centroid_x'] = centers[:, 0].mean()
            metrics['density_std'] = centers.std()
            self.history.append(metrics['density_std'])

        # 异常检测：密度标准差突变（鱼群聚集/逃散行为）
        metrics['anomaly'] = self._detect_anomaly(metrics)
        return metrics

    def _detect_anomaly(self, metrics: dict) -> str:
        if len(self.history) < 10:
            return 'NORMAL'
        recent_avg = np.mean(list(self.history)[-5:])
        hist_avg   = np.mean(list(self.history)[:-5])
        if recent_avg > hist_avg * self.alert_threshold:
            return 'ALERT_CROWDING'
        if metrics.get('fish_count', 0) < 3:
            return 'ALERT_LOW_COUNT'
        return 'NORMAL'
