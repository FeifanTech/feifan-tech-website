  • 算法调参建议可能不适合你的数据
  • 生成的特征可能存在数据泄露
  • 代码性能优化建议需要实测验证

Python 销量预测特征工程 — AI协助生成并人工校验版 
import pandas as pd
import numpy as np

def build_sales_features(df: pd.DataFrame) -> pd.DataFrame:
    """
    销量预测特征工程
    注意：所有特征基于历史数据计算，避免时序泄露
    df 必须按 (sku_id, date) 排序
    """
    df = df.sort_values(['sku_id', 'date']).copy()

    # ── 1. 时间特征（无泄露） ──
    df['dayofweek']   = df['date'].dt.dayofweek
    df['month']       = df['date'].dt.month
    df['day_of_year'] = df['date'].dt.dayofyear
    df['is_weekend']  = df['dayofweek'].isin([5, 6]).astype(int)
    df['week_of_year'] = df['date'].dt.isocalendar().week

    # ── 2. 滞后特征（关键：避免泄露，使用shift） ──
    lags = [1, 7, 14, 28, 365]
    for lag in lags:
        df[f'sales_lag_{lag}'] = (
            df.groupby('sku_id')['sales_qty']
              .shift(lag)  # ⚠️ shift避免泄露
        )

    # ── 3. 滚动统计特征 ──
    windows = [7, 14, 28]
    for w in windows:
        rolling = (
            df.groupby('sku_id')['sales_qty']
              .transform(lambda x: x.shift(1).rolling(w, min_periods=1).mean())
        )
        df[f'sales_roll_mean_{w}'] = rolling

    # ── 4. 促销/节假日增益特征 ──
    df['promo_x_holiday'] = df['promo_type'].notna().astype(int) * df['is_holiday']

    # ── 5. SKU生命周期特征 ──
    df['sku_age_days'] = (
        df.groupby('sku_id')['date']
          .transform(lambda x: (x - x.min()).dt.days)
    )
    df['is_new_sku'] = (df['sku_age_days'] < 30).astype(int)

    feature_cols = [c for c in df.columns if c not in ['date', 'sales_qty']]
    print(f"生成特征数量: {len(feature_cols)}")
    return df
⚠️
时序泄露是最常见的致命错误
AI生成的特征工程代码有时会忘记使用 shift(1)，导致用到未来数据（数据泄露）。结果是线下AUC超高（0.99+）但线上效果极差。务必检查所有滚动/聚合特征是否正确使用了shift。

import optuna
import lightgbm as lgb
from sklearn.model_selection import StratifiedKFold
from sklearn.metrics import precision_score, roc_auc_score
import numpy as np
optuna.logging.set_verbosity(optuna.logging.WARNING)

def objective(trial):
    params = {
        'objective': 'binary',
        'metric': 'auc',
        'verbosity': -1,
        # 不均衡处理：正负样本比约1:9
        'scale_pos_weight': 9,
        # Optuna建议的超参空间
        'n_estimators': trial.suggest_int('n_estimators', 100, 1000),
        'learning_rate': trial.suggest_float('learning_rate', 0.01, 0.3, log=True),
        'max_depth': trial.suggest_int('max_depth', 3, 10),
        'num_leaves': trial.suggest_int('num_leaves', 20, 200),
        'min_child_samples': trial.suggest_int('min_child_samples', 10, 100),
        'subsample': trial.suggest_float('subsample', 0.6, 1.0),
        'colsample_bytree': trial.suggest_float('colsample_bytree', 0.6, 1.0),
        'reg_alpha': trial.suggest_float('reg_alpha', 1e-8, 10.0, log=True),
        'reg_lambda': trial.suggest_float('reg_lambda', 1e-8, 10.0, log=True),
    }

🐛 常见错误快速排查
  • • MemoryError → 问AI如何分批处理
  • • ValueError: shape mismatch → 让AI检查特征对齐
  • • 线下好线上差 → 让AI检查数据泄露
  • • 训练极慢 → 让AI推荐并行/向量化方案
  • • 推理慢 → 问AI模型压缩/缓存方案
⚡ 性能优化策略
  • • 用 pandas 向量化替代for循环
  • • 特征工程用 groupby.transform
  • • LightGBM使用 n_jobs=-1 并行
  • • 大数据用 Dask/Polars 替代pandas
  • • 推理用 ONNX 加速模型部署
